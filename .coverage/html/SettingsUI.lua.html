7 local function GetProfile()
8 FlightsimDB = FlightsimDB or {}
9 FlightsimDB.profile = FlightsimDB.profile or {}
10 local p = FlightsimDB.profile
11
12 p.ui = p.ui or {}
13 p.speedBar = p.speedBar or {}
14 p.visibility = p.visibility or {}
15 p.abilityBars = p.abilityBars or {}
16
17 -- General
18 if p.locked == nil then
19 p.locked = false
20 end
21 if p.scale == nil then
22 p.scale = 1
23 end
24
25 -- UI dimensions
26 if p.ui.width == nil then
27 p.ui.width = 150
28 end
29 if p.ui.speedBarHeight == nil then
30 p.ui.speedBarHeight = 20
31 end
32 if p.ui.sustainableSpeedMarkerWidth == nil then
33 p.ui.sustainableSpeedMarkerWidth = p.ui.optimalMarkerWidth or 1
34 end
35 if p.ui.sustainableSpeedMarkerAlpha == nil then
36 p.ui.sustainableSpeedMarkerAlpha = 0.2
37 end
38 if p.ui.accelBarHeight == nil then
39 p.ui.accelBarHeight = 1
40 end
41 if p.ui.accelBarGap == nil then
42 p.ui.accelBarGap = 0
43 end
44 if p.ui.abilityBarHeight == nil then
45 p.ui.abilityBarHeight = 5
46 end
47 if p.ui.barGap == nil then
48 p.ui.barGap = 2
49 end
50
51 -- Speed bar
52 if p.speedBar.maxSpeed == nil then
53 p.speedBar.maxSpeed = 950
54 end
55 if p.speedBar.sustainableSpeed == nil then
56 p.speedBar.sustainableSpeed = p.speedBar.optimalSpeed or 790
57 end
58 if p.speedBar.fontSize == nil then
59 p.speedBar.fontSize = 10
60 end
61 if p.speedBar.showPercent == nil then
62 p.speedBar.showPercent = true
63 end
64
65 -- Visibility
66 if p.visibility.hideWhenNotSkyriding == nil then
67 p.visibility.hideWhenNotSkyriding = true
68 end
69
70 -- Ability bars
71 if p.abilityBars.showSurgeForward == nil then -- luacheck: ignore
72 p.abilityBars.showSurgeForward = true
73 end
74 if p.abilityBars.showSecondWind == nil then
75 p.abilityBars.showSecondWind = true
76 end
77 if p.abilityBars.showWhirlingSurge == nil then
78 p.abilityBars.showWhirlingSurge = true
79 end
80
81 return p
82 end

87 return ok
