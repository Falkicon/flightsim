130 local L = Flightsim.L
131 msg = (msg or ""):lower():gsub("^%s+", ""):gsub("%s+$", "")

133 if not FlightsimDB or not FlightsimDB.profile then
134 print(L["NOT_INITIALIZED_YET"])

137
138 if msg == "lock" then
139 FlightsimDB.profile.locked = true
140 print(L["LOCKED"])
141 return
142 elseif msg == "unlock" then
143 FlightsimDB.profile.locked = false
144 print(L["UNLOCKED"])
145 return
146 elseif msg:match("^scale%s+") then
147 local val = tonumber(msg:match("^scale%s+([%d%.]+)$"))
148 if not val then
149 print(L["USAGE_SCALE"])
150 return
151 end
152 if FlightsimUI and FlightsimUI.SetScale then
153 FlightsimUI:SetScale(val)
154 end
155 print(L["SCALE_SET"])
156 return
157 elseif msg:match("^width%s+") then
158 local val = tonumber(msg:match("^width%s+([%d%.]+)$"))
159 if not val then
160 print(L["USAGE_WIDTH"])
161 return
162 end
163 if FlightsimUI and FlightsimUI.SetWidth then
164 FlightsimUI:SetWidth(val)
165 end
166 print(L["WIDTH_SET"])
167 return
168 elseif msg:match("^barmax%s+") then
169 local val = tonumber(msg:match("^barmax%s+([%d%.]+)$"))
170 if not val then
171 print(L["USAGE_BARMAX"])
172 return
173 end
174 if FlightsimUI and FlightsimUI.SetSpeedBarMax then
175 FlightsimUI:SetSpeedBarMax(val)
176 end
177 print(L["BARMAX_SET"])
178 return
179 elseif msg:match("^sustainable%s+") or msg:match("^optimal%s+") then
180 local val = tonumber(msg:match("^sustainable%s+([%d%.]+)$") or msg:match("^optimal%s+([%d%.]+)$"))
181 if val == nil then
182 print(L["USAGE_SUSTAINABLE"])
183 return
184 end
185 if FlightsimUI and FlightsimUI.SetSustainableSpeed then
186 FlightsimUI:SetSustainableSpeed(val)
187 end
188 print(L["SUSTAINABLE_SET"])
189 return
190 elseif msg == "hidenot" then
191 FlightsimDB.profile.visibility.hideWhenNotSkyriding = true
192 FlightsimDB.profile.visibility.hideWhileSkyriding = false
193 print(L["HIDE_NOT_SKYRIDING"])
194 return
195 elseif msg == "hidesky" then
196 FlightsimDB.profile.visibility.hideWhileSkyriding = true
197 FlightsimDB.profile.visibility.hideWhenNotSkyriding = false
198 print(L["HIDE_WHILE_SKYRIDING"])
199 return
200 elseif msg == "showalways" then
201 FlightsimDB.profile.visibility.hideWhileSkyriding = false
202 FlightsimDB.profile.visibility.hideWhenNotSkyriding = false
203 print(L["SHOW_ALWAYS"])
204 return
205 elseif msg:match("^toggle%s+") then
206 local query = msg:match("^toggle%s+(.+)$")
207 local found
208 for _, token in ipairs(FlightsimDB.profile.abilities.order or {}) do
209 if token:lower():find(query, 1, true) then
210 local enabled = FlightsimDB.profile.abilities.enabled[token] ~= false
211 FlightsimDB.profile.abilities.enabled[token] = not enabled
212 print(string.format(L["ABILITY_TOGGLED"], token, (not enabled) and "enabled" or "disabled"))
213 found = true
214 break
215 end
216 end
217 if not found then
218 print(L["ABILITY_NOT_FOUND"])
219 else
220 if FlightsimUI and FlightsimUI.RebuildAbilityRows then
221 FlightsimUI:RebuildAbilityRows()
222 end
223 end
224 return
225 elseif msg == "list" then
226 print(L["ABILITIES_LIST"])
227 for i, token in ipairs(FlightsimDB.profile.abilities.order or {}) do
228 local enabled = FlightsimDB.profile.abilities.enabled[token] ~= false
229 print(string.format(" %d. %s [%s]", i, token, enabled and "ON" or "OFF"))
230 end
231 return
232 elseif msg:match("^move%s+") then
233 local a, b = msg:match("^move%s+(.+)%s+(%d+)$")
234 local newIndex = tonumber(b)
235 if not (a and newIndex) then
236 print(L["USAGE_MOVE"])
237 return
238 end
239 local order = FlightsimDB.profile.abilities.order or {}
240 local fromIndex
241 for i, token in ipairs(order) do
242 if token:lower():find(a, 1, true) then
243 fromIndex = i

245 end
246 end
247 if not fromIndex then
248 print(L["ABILITY_NOT_FOUND"])
249 return
250 end
251 newIndex = math.max(1, math.min(#order, newIndex))
252 local item = table.remove(order, fromIndex)
253 table.insert(order, newIndex, item)
254 FlightsimDB.profile.abilities.order = order
255 if FlightsimUI and FlightsimUI.RebuildAbilityRows then
256 FlightsimUI:RebuildAbilityRows()
257 end
258 print(L["ORDER_UPDATED"])
259 return
260 elseif msg == "debug" then
261 FlightsimDB.profile.debugMode = not FlightsimDB.profile.debugMode
262 Flightsim.debugMode = FlightsimDB.profile.debugMode
263 print("Flightsim debug mode:", Flightsim.debugMode and "ON" or "OFF")
264 return
265 elseif msg == "status" then
266 if FlightsimUI and FlightsimUI.Status then
267 FlightsimUI:Status()
268 else
269 print(L["STATUS_NOT_AVAILABLE"])
270 end
271 return
272 elseif msg == "reset" then
273 FlightsimDB.profile.x = 0
274 FlightsimDB.profile.y = 0
275 FlightsimDB.profile.scale = 1
276 if FlightsimUI and FlightsimUI.frame then
277 FlightsimUI.frame:ClearAllPoints()

279 FlightsimUI.frame:SetScale(1)
280 end
